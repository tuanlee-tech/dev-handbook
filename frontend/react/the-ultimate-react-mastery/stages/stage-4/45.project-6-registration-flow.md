## 🤔 Kiểm tra đầu vào (5 phút)

1. **React Hook Form useForm hook có những options nào quan trọng?**
   - _Đáp án: `resolver` (tích hợp Zod/Yup), `defaultValues` (giá trị khởi tạo), `mode` (khi nào validate: onChange/onBlur/onSubmit/all), `criteriaMode` (hiển thị lỗi: firstError/allErrors)._

2. **Zod schema composition hoạt động như thế nào?**
   - _Đáp án: Dùng `.merge()`, `.extend()`, `.pick()`, `.omit()`, `.partial()`, `.required()` để kết hợp/tái sử dụng schemas. Ví dụ: `baseSchema.extend({ newField: z.string() })` hoặc `baseSchema.pick({ field1: true, field2: true })` để tạo schema con._

3. **Context API pattern nào tốt nhất cho form state management?**
   - _Đáp án: Pattern Provider + Custom Hook (`useFormContext`) + state object tập trung. Tránh dùng nhiều Context riêng lẻ, nên gom state vào 1 object để giảm re-render. Nếu phức tạp thì dùng `useReducer` thay vì `useState` nhiều cái._

import { useState, createContext, useContext } from 'react';
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';

// ============= VALIDATION SCHEMAS =============

const personalInfoSchema = z.object({
  firstName: z
    .string()
    .min(2, 'First name must be at least 2 characters')
    .max(50, 'First name too long'),
  lastName: z
    .string()
    .min(2, 'Last name must be at least 2 characters')
    .max(50, 'Last name too long'),
  birthDate: z.string().refine((date) => {
    const age = new Date().getFullYear() - new Date(date).getFullYear();
    return age >= 18;
  }, 'Must be at least 18 years old'),
  gender: z.enum(['male', 'female', 'other', 'prefer-not-to-say']),
});

const contactSchema = z.object({
  email: z.string().email('Invalid email address').min(1, 'Email is required'),
  phone: z
    .string()
    .regex(/^\+?[1-9]\d{9,14}$/, 'Invalid phone number')
    .min(1, 'Phone is required'),
  address: z.string().min(5, 'Address too short').max(200, 'Address too long'),
  city: z.string().min(2, 'City required'),
  country: z.string().min(2, 'Country required'),
});

const accountSchema = z
  .object({
    username: z
      .string()
      .min(3, 'Username must be at least 3 characters')
      .max(20, 'Username too long')
      .regex(
        /^[a-zA-Z0-9_]+$/,
        'Only letters, numbers, and underscore allowed',
      ),
    password: z
      .string()
      .min(8, 'Password must be at least 8 characters')
      .regex(/[A-Z]/, 'Must contain uppercase letter')
      .regex(/[a-z]/, 'Must contain lowercase letter')
      .regex(/[0-9]/, 'Must contain number')
      .regex(/[^A-Za-z0-9]/, 'Must contain special character'),
    confirmPassword: z.string(),
  })
  .refine((data) => data.password === data.confirmPassword, {
    message: "Passwords don't match",
    path: ['confirmPassword'],
  });

const preferencesSchema = z.object({
  newsletter: z.boolean(),
  notifications: z.enum(['all', 'important', 'none']),
  language: z.enum(['en', 'vi', 'fr', 'es']),
  timezone: z.string().min(1, 'Timezone required'),
});

// ============= FORM CONTEXT =============

const FormContext = createContext();

function FormProvider({ children }) {
  const [formData, setFormData] = useState({
    personalInfo: {},
    contact: {},
    account: {},
    preferences: {},
  });

  const [currentStep, setCurrentStep] = useState(0);

  const updateFormData = (step, data) => {
    setFormData((prev) => ({
      ...prev,
      [step]: { ...prev[step], ...data },
    }));
  };

  const nextStep = () => setCurrentStep((prev) => Math.min(prev + 1, 4));
  const prevStep = () => setCurrentStep((prev) => Math.max(prev - 1, 0));
  const goToStep = (step) => setCurrentStep(step);

  return (
    <FormContext.Provider
      value={{
        formData,
        updateFormData,
        currentStep,
        nextStep,
        prevStep,
        goToStep,
      }}
    >
      {children}
    </FormContext.Provider>
  );
}

function useFormContext() {
  const context = useContext(FormContext);
  if (!context) {
    throw new Error('useFormContext must be used within FormProvider');
  }
  return context;
}

// ============= PROGRESS INDICATOR =============

function ProgressBar({ currentStep, totalSteps }) {
  const progress = ((currentStep + 1) / totalSteps) * 100;

  return (
    <div style={{ marginBottom: 32 }}>
      {/* Progress bar */}
      <div
        style={{
          width: '100%',
          height: 8,
          backgroundColor: '#e5e7eb',
          borderRadius: 4,
          overflow: 'hidden',
        }}
      >
        <div
          style={{
            width: `${progress}%`,
            height: '100%',
            backgroundColor: '#3b82f6',
            transition: 'width 0.3s ease',
          }}
        />
      </div>

      {/* Step indicators */}
      <div
        style={{
          display: 'flex',
          justifyContent: 'space-between',
          marginTop: 16,
        }}
      >
        {['Personal', 'Contact', 'Account', 'Preferences', 'Review'].map(
          (label, idx) => (
            <div
              key={idx}
              style={{
                flex: 1,
                textAlign: 'center',
                fontSize: 12,
                color: idx <= currentStep ? '#3b82f6' : '#9ca3af',
                fontWeight: idx === currentStep ? 'bold' : 'normal',
              }}
            >
              <div
                style={{
                  width: 32,
                  height: 32,
                  borderRadius: '50%',
                  backgroundColor: idx <= currentStep ? '#3b82f6' : '#e5e7eb',
                  color: idx <= currentStep ? 'white' : '#6b7280',
                  display: 'flex',
                  alignItems: 'center',
                  justifyContent: 'center',
                  margin: '0 auto 8px',
                  fontWeight: 'bold',
                }}
              >
                {idx + 1}
              </div>
              {label}
            </div>
          ),
        )}
      </div>
    </div>
  );
}

          backgroundColor: isFirstStep ? '#e5e7eb' : 'white',
          border: '1px solid #d1d5db',

          fontSize: 16,

        onClick={onNext}

          backgroundColor: isValid ? '#3b82f6' : '#93c5fd',
          color: 'white',
          border: 'none',
          borderRadius: 8,
          cursor: isValid ? 'pointer' : 'not-allowed',
          fontSize: 16,
          fontWeight: 600,
        }}
      >
        {isLastStep ? 'Submit' : 'Next →'}
      </button>
    </div>
  );
}

function Step3_Account() {
  const { formData, updateFormData, nextStep, prevStep } = useFormContext();

  const {
    register,
    handleSubmit,
    formState: { errors, isValid },
    watch,
  } = useForm({
    resolver: zodResolver(accountSchema),
    defaultValues: formData.account,
    mode: 'onChange',
  });

  const password = watch('password');

  const onSubmit = (data) => {
    updateFormData('account', data);
    nextStep();
  };

  // Password strength indicator
  const getPasswordStrength = (pwd) => {
    if (!pwd) return { strength: 0, label: 'No password' };
    let strength = 0;
    if (pwd.length >= 8) strength++;
    if (/[A-Z]/.test(pwd)) strength++;
    if (/[a-z]/.test(pwd)) strength++;
    if (/[0-9]/.test(pwd)) strength++;
    if (/[^A-Za-z0-9]/.test(pwd)) strength++;

    const labels = ['Very Weak', 'Weak', 'Fair', 'Good', 'Strong'];
    return { strength, label: labels[strength - 1] || 'Very Weak' };
  };

  const passwordStrength = getPasswordStrength(password);

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <h2 style={{ marginBottom: 24 }}>Account Details</h2>

      <div style={{ marginBottom: 16 }}>
        <label style={{ display: 'block', marginBottom: 8, fontWeight: 500 }}>
          Username *
        </label>
        <input
          {...register('username')}
          placeholder='Choose a username'
          style={{
            width: '100%',
            padding: 12,
            border: `1px solid ${errors.username ? '#dc2626' : '#d1d5db'}`,
            borderRadius: 8,
            fontSize: 16,
          }}
        />
        {errors.username && (
          <p style={{ color: '#dc2626', fontSize: 14, marginTop: 4 }}>
            {errors.username.message}
          </p>
        )}
      </div>

      <div style={{ marginBottom: 16 }}>
        <label style={{ display: 'block', marginBottom: 8, fontWeight: 500 }}>
          Password *
        </label>
        <input
          type='password'
          {...register('password')}
          placeholder='Create a strong password'
          style={{
            width: '100%',
            padding: 12,
            border: `1px solid ${errors.password ? '#dc2626' : '#d1d5db'}`,
            borderRadius: 8,
            fontSize: 16,
          }}
        />

        {/* Password strength indicator */}
        {password && (
          <div style={{ marginTop: 8 }}>
            <div style={{ display: 'flex', gap: 4, marginBottom: 4 }}>
              {[1, 2, 3, 4, 5].map((level) => (
                <div
                  key={level}
                  style={{
                    flex: 1,
                    height: 4,
                    backgroundColor:
                      level <= passwordStrength.strength
                        ? passwordStrength.strength < 3
                          ? '#dc2626'
                          : passwordStrength.strength < 4
                            ? '#d97706'
                            : '#059669'
                        : '#e5e7eb',
                    borderRadius: 2,
                  }}
                />
              ))}
            </div>
            <p style={{ fontSize: 12, color: '#6b7280' }}>
              Strength: {passwordStrength.label}
            </p>
          </div>
        )}

        {errors.password && (
          <p style={{ color: '#dc2626', fontSize: 14, marginTop: 4 }}>
            {errors.password.message}
          </p>
        )}
      </div>

      <div style={{ marginBottom: 16 }}>
        <label style={{ display: 'block', marginBottom: 8, fontWeight: 500 }}>
          Confirm Password *
        </label>
        <input
          type='password'
          {...register('confirmPassword')}
          placeholder='Re-enter password'
          style={{

/**
 * Auto-save to localStorage
 */

function FormProvider({ children }) {
  const [formData, setFormData] = useState(() => {
    // Load from localStorage on init
    const saved = localStorage.getItem('registrationFormData');
    return saved
      ? JSON.parse(saved)
      : {
          personalInfo: {},
          contact: {},
          account: {},
          preferences: {},
        };
  });

  const [currentStep, setCurrentStep] = useState(() => {
    const savedStep = localStorage.getItem('registrationStep');
    return savedStep ? parseInt(savedStep) : 0;
  });

  const updateFormData = (step, data) => {
    setFormData((prev) => {
      const updated = {
        ...prev,
        [step]: { ...prev[step], ...data },
      };

      // Save to localStorage
      localStorage.setItem('registrationFormData', JSON.stringify(updated));

      return updated;
    });
  };

  const nextStep = () => {
    setCurrentStep((prev) => {
      const next = Math.min(prev + 1, 4);
      localStorage.setItem('registrationStep', next.toString());
      return next;
    });
  };

  const prevStep = () => {
    setCurrentStep((prev) => {
      const next = Math.max(prev - 1, 0);
      localStorage.setItem('registrationStep', next.toString());
      return next;
    });
  };

  const clearForm = () => {
    localStorage.removeItem('registrationFormData');
    localStorage.removeItem('registrationStep');
    setFormData({
      personalInfo: {},
      contact: {},
      account: {},
      preferences: {},
    });
    setCurrentStep(0);
  };

  return (
    <FormContext.Provider
      value={{
        formData,
        updateFormData,
        currentStep,
        nextStep,
        prevStep,
        goToStep: setCurrentStep,
        clearForm,
      }}
    >
      {children}
    </FormContext.Provider>
  );
}

// ✅ Form data persisted across browser refreshes
// ✅ User can continue where they left off
// ✅ Clear function to reset everything

/** Danh sách domain email doanh nghiệp được phép đăng ký */
const ALLOWED_EMAIL_DOMAINS = [
  'company.com',
  'enterprise.vn',
  'corp.io',
  'tech.dev',
];

/** Danh sách mã quốc gia hợp lệ cho số điện thoại */
const ALLOWED_COUNTRY_CODES = ['+1', '+84', '+33', '+34', '+44', '+49', '+61'];

/** Cấu hình độ mạnh mật khẩu - có thể tùy chỉnh dễ dàng */
const PASSWORD_STRENGTH_CONFIG = {
  minLength: 8,
  requireUppercase: true,
  requireLowercase: true,
  requireNumber: true,
  requireSpecialChar: true,
};

/**
 * Kiểm tra email có thuộc domain doanh nghiệp không
 */
const isCorporateEmail = (email) => {
  if (!email || !email.includes('@')) return false;
  const domain = email.split('@')[1]?.toLowerCase();
  return ALLOWED_EMAIL_DOMAINS.includes(domain);
};

/**
 * Kiểm tra số điện thoại có mã quốc gia hợp lệ không
 */
const hasValidCountryCode = (phone) => {
  if (!phone) return false;
  return ALLOWED_COUNTRY_CODES.some((code) => phone.startsWith(code));
};

/**
 * Kiểm tra độ mạnh mật khẩu dựa trên cấu hình tùy chỉnh
 * Trả về { valid: boolean, strength: number, errors: string[] }
 */
const checkPasswordStrength = (password) => {
  if (!password) return { valid: false, strength: 0, errors: [] };
  let strength = 0;
  const errors = [];

  if (password.length >= PASSWORD_STRENGTH_CONFIG.minLength) strength++;
  else errors.push(`Ít nhất ${PASSWORD_STRENGTH_CONFIG.minLength} ký tự`);

  if (PASSWORD_STRENGTH_CONFIG.requireUppercase && /[A-Z]/.test(password))
    strength++;
  else if (PASSWORD_STRENGTH_CONFIG.requireUppercase)
    errors.push('Chứa chữ hoa (A-Z)');

  if (PASSWORD_STRENGTH_CONFIG.requireLowercase && /[a-z]/.test(password))
    strength++;
  else if (PASSWORD_STRENGTH_CONFIG.requireLowercase)
    errors.push('Chứa chữ thường (a-z)');

  if (PASSWORD_STRENGTH_CONFIG.requireNumber && /[0-9]/.test(password))
    strength++;
  else if (PASSWORD_STRENGTH_CONFIG.requireNumber) errors.push('Chứa số (0-9)');

  if (
    PASSWORD_STRENGTH_CONFIG.requireSpecialChar &&
    /[^A-Za-z0-9]/.test(password)
  )
    strength++;
  else if (PASSWORD_STRENGTH_CONFIG.requireSpecialChar)
    errors.push('Chứa ký tự đặc biệt (!@#$...)');

  return { valid: errors.length === 0, strength, errors };
};

/**
 * Giả lập gọi API kiểm tra username đã tồn tại chưa (async)
 * Trong thực tế: fetch(`/api/check-username?username=${username}`)

    return () => updateTotalTimeSpent(startedAt);
  }, [currentStep]);

  /** Theo dõi drop-off khi rời trang */
  useEffect(() => {
    const handleBeforeUnload = () => {
      trackAnalytics(currentStep, 'dropoff');
    };
    window.addEventListener('beforeunload', handleBeforeUnload);
    return () => window.removeEventListener('beforeunload', handleBeforeUnload);
  }, [currentStep]);

  // Kiểm tra query param ?resume=token khi mount
  useEffect(() => {
    const params = new URLSearchParams(window.location.search);
    const resume = params.get('resume');
    const savedToken = localStorage.getItem(RESUME_TOKEN_KEY);
    if (!resume || resume !== savedToken) return;

    // chỉ schedule load sau khi effect hoàn tất
    queueMicrotask(() => {
      loadDraft();

      window.history.replaceState({}, document.title, window.location.pathname);
    });
  }, [loadDraft]);

  return (
    <FormContext.Provider
      value={{
        formData,
        updateFormData,
        currentStep,
        nextStep,
        prevStep,
        goToStep,
        clearForm,
        // --- Các giá trị mới ---
        isDraft,
        lastSaved,
        saveDraft,
        sendResumeLink,
        draftLink,
        resumeToken,
        abVariant,
        getCompletionRates,
      }}
    >
      {children}
    </FormContext.Provider>
  );
}

### Câu Hỏi Phỏng Vấn

**Junior Level:**

1. **Q:** Multi-step form có lợi ích gì so với single-page form?

   **A:** Giảm cognitive load cho user, tăng completion rate với progress indicator, cho phép grouping logic (personal → contact → account), validate từng phần thay vì 1 lần submit, dễ implement conditional fields và save-as-draft.

2. **Q:** Làm sao persist data across steps?

   **A:** Dùng Context/State kết hợp `defaultValues` của react-hook-form cho session persist; localStorage/sessionStorage cho cross-session; URL query params hoặc session token cho resume later / shareable link.

3. **Q:** Validation từng step hoạt động như thế nào?

   **A:** Mỗi step có Zod/Yup schema riêng chỉ chứa fields của step đó; gọi `handleSubmit` trước khi `nextStep()` để validate; dùng `mode: 'onChange'` để feedback real-time; lỗi chỉ hiển thị fields trong step hiện tại.

**Mid Level:**

1. **Q:** So sánh approaches khác nhau cho multi-step forms?

   **A:**
   - (1) _State-driven_ (React state/Context): đơn giản, nhưng khó share URL;
   - (2) _URL-driven_ (route per step): cho phép back/forward browser, refresh giữ nguyên step, SEO-friendly;
   - (3) _State machine_ (XState): xử lý tốt conditional branching, skip logic, guard transitions;
   - (4) _Compound components_: flexible UI nhưng phức tạp hơn.

2. **Q:** Optimize performance trong wizard forms?

   **A:** `React.memo` + `useMemo` cho từng step component để tránh re-render; lazy load/unmount steps không active (trade-off với keep-alive state); debounce async validation (username check); code-splitting từng step bằng `React.lazy`; tránh đưa toàn bộ form state vào Context nếu không cần thiết.

3. **Q:** Handle errors trong multi-step submissions?

   **A:** Implement _partial save_ — lưu draft trước khi submit; nếu API lỗi ở step N, navigate user về step đó với error message; dùng Error Boundary cho từng step; retry mechanism với exponential backoff cho network errors; rollback optimistic updates nếu submit thất bại.

**Senior Level:**

1. **Q:** Design system architecture cho complex form flows?

   **A:** Dùng _schema-driven forms_ — JSON config định nghĩa fields, validation, dependencies; _state machine_ (XState) quản lý flow transitions, guards, và conditional branching; _event-driven architecture_ — form emits events, backend orchestrates next step; _micro-frontend_ per step nếu teams sở hữu riêng; plugin system cho custom step types mà không sửa core.

2. **Q:** A/B testing strategies cho form optimization?

   **A:** Server-driven flow configuration qua feature flags (LaunchDarkly/Firebase); define variants ở step level (thứ tự fields, số lượng bước, button copy); track _step completion rate_ và _time-on-step_ per variant; cần đủ sample size để đạt statistical significance (thường >1000 users/variant); cohort analysis để so sánh long-term conversion, không chỉ form submission.

3. **Q:** Analytics và conversion tracking trong forms?

   **A:** Emit custom events mỗi lần `nextStep`/`prevStep`/`submit` với timestamp; đo _drop-off funnel_ per step để identify bottleneck; track _field interaction_ (focus time, correction count) để phát hiện confusing fields; correlation analysis giữa field order và abandonment rate; integrate với product analytics (Mixpanel/Amplitude) hoặc GA4 với custom dimensions cho form variant ID.

### War Stories

**Story: The 20-Step Monster**

"Client yêu cầu onboarding form với 20 steps. Users abandoned massively (85% drop-off).

Solution:

- Reduced to 5 essential steps
- Made 10 steps optional (edit profile later)
- Removed 5 steps entirely (not needed)
- Added progress save every step

Result: Completion rate từ 15% → 65%

Lesson: Less is more. Only ask what you NEED, not what you WANT."
